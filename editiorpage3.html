{

int t = in.nextInt();
while(t--!=0)
{ int n=in.nextInt();
int k=in.nextInt();
int x=in.nextInt();
int y=in.nextInt();
int i=x;
boolean flag=false;
if(x==y)
flag=true;
x=(x+k)%n;
while(x!=i)
{ if(x==y)
flag=true;
x=(x+k)%n;
}
if(flag==true)
System.out.println("YES");
else
System.out.println("NO");
}
}
